USE kastoria;

CREATE TABLE Students(
    St_AM INT(4) NOT NULL,
    St_Name VARCHAR(20) NOT NULL,
    St_Surname VARCHAR(20) NOT NULL,
    PRIMARY KEY(St_Am)
);

CREATE TABLE StudentsFree(
    St_ID INT(5000) NOT NULL,
    St_Free_Name VARCHAR(20),
    St_Free_Sur VARCHAR(20),
    St_AM INT(4) NOT NULL,
    PRIMARY KEY(St_ID),
    FOREIGN KEY(St_AM) REFERENCES Students(St_AM)
); 

CREATE TABLE StudentsNotFree(
    St_Not_Free_ID INT(5000) NOT NULL,
    St_Not_Free_Name VARCHAR(20),
    St_Not_Free_Sur VARCHAR(20),
    St_AM INT(4) NOT NULL,
    PRIMARY KEY(St_Not_Free_ID),
    FOREIGN KEY(St_AM) REFERENCES Students(St_AM)
);
